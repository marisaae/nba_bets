CREATE TABLE team_info (id INT PRIMARY KEY, team_name VARCHAR(50), abbreviation VARCHAR(10), location VARCHAR(50), total_record VARCHAR(10), total_win INT, total_loss INT, home_record VARCHAR(10), home_win INT, home_loss INT, away_record VARCHAR(10), away_win INT, away_loss INT, standing INT, logo VARCHAR(255), last_updated TIMESTAMP DEFAULT NOW());

CREATE TABLE roster (player_id INT PRIMARY KEY, team_id INT, full_name VARCHAR(50), first_name VARCHAR(25), last_name VARCHAR(25), age INT, number INT, position VARCHAR(10), height VARCHAR (10), weight VARCHAR(10), last_updated TIMESTAMP DEFAULT NOW());

CREATE TABLE schedule (game_id INT PRIMARY KEY, game_label VARCHAR(50), season_year VARCHAR(10), game_date TIMESTAMP, game_status VARCHAR(25), home_team_id INT, home_team_name VARCHAR(25), away_team_id INT, away_team_name VARCHAR(25), arena_name VARCHAR(50), arena_city VARCHAR(50), arena_state VARCHAR(10), last_updated TIMESTAMP DEFAULT NOW());

CREATE TABLE player_game_log (player_id INT, game_id INT, game_date VARCHAR(25), matchup VARCHAR(25), wl CHAR(1), min INT, pts INT, fgm INT, fga INT, fg_pct FLOAT, three_pts_made INT, three_pts_att INT, three_pts_pct FLOAT, ftm INT, fta INT, ft_pct FLOAT, oreb INT, dreb INT, tot_reb INT, ast INT, stl INT, blk INT, turnover INT, fouls INT, last_updated TIMESTAMP DEFAULT NOW(), PRIMARY KEY(game_id, player_id));


CREATE TABLE game_odds (id SERIAL PRIMARY KEY, event_id VARCHAR(50) NOT NULL, bookmaker VARCHAR(25) NOT NULL, market VARCHAR (10) NOT NULL, outcome_name VARCHAR (50), price FLOAT, point FLOAT, last_odds_update TIMESTAMP, last_updated TIMESTAMP DEFAULT NOW(), UNIQUE(event_id, bookmaker, market, outcome_name));

CREATE TABLE player_odds (id SERIAL PRIMARY KEY, event_id VARCHAR(50) NOT NULL, bookmaker VARCHAR(25) NOT NULL, market VARCHAR(10) NOT NULL, outcome_name VARCHAR (50), player_name VARCHAR(50) NOT NULL, price FLOAT, point FLOAT, last_odds_update TIMESTAMP, last_updated TIMESTAMP DEFAULT NOW(), UNIQUE(event_id, bookmaker, market, outcome_name, player_name));
